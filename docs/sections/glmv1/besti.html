ai <-  read.csv("inventory.csv")
# total yearly sales for each item
inventory_sum <- ai %>%
  group_by(item_no) %>%
  summarise(year_total = sum(sold), .groups = "drop")

# Join this back to the original data frame
ai <- left_join(ai, inventory_sum, by = c("item_no"))
# Calculate the top 10% threshold
top_10_threshold <- quantile(ai$year_total, 0.9)

# Create the 'hv_product' column
ai <- ai %>% 
  mutate(best_seller = ifelse(year_total >= top_10_threshold, 1, 0))
ai$year <-  2023
ai <- ai %>%
  mutate(MMWRyear = ifelse(week == 0, year - 1, ifelse(week == 53, year + 1, year)),
         week = ifelse(week == 0, 52, ifelse(week == 53, 1, week)))

ai$date <- MMWRweek::MMWRweek2Date(ai$MMWRyear, ai$week)
ai$month <- month(ai$date, label = TRUE)
ai$month <- month(ai$date)
ai <-  ai %>% 
  select("year" = MMWRyear,
         item_no,
         month,
         week,
         best_seller,
         sold)
